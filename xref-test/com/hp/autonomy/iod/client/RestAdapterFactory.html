6   package com.hp.autonomy.iod.client;

8   import com.hp.autonomy.iod.client.converter.IodConverter;
9   import com.hp.autonomy.iod.client.error.IodErrorHandler;
10  import com.hp.autonomy.iod.client.util.ApiKeyRequestInterceptor;
11  import org.apache.http.HttpHost;
12  import org.apache.http.impl.client.HttpClientBuilder;
13  import retrofit.RestAdapter;
14  import retrofit.client.ApacheClient;
15  import retrofit.converter.JacksonConverter;
16  
17  public class RestAdapterFactory {
18  
19      public static RestAdapter getRestAdapter(final boolean withInterceptor) {
20          return getRestAdapter(withInterceptor, Endpoint.PRODUCTION);
21      }
22  
23      public static RestAdapter getRestAdapter(final boolean withInterceptor, final Endpoint endpoint) {
24          final HttpClientBuilder builder = HttpClientBuilder.create();
25  
26          final String proxyHost = System.getProperty("hp.iod.https.proxyHost");
27  
28          if (proxyHost != null) {
29              final Integer proxyPort = Integer.valueOf(System.getProperty("hp.iod.https.proxyPort", "8080"));
30              builder.setProxy(new HttpHost(proxyHost, proxyPort));
31          }
32  
33          final RestAdapter.Builder restAdapterBuilder = new RestAdapter.Builder()
34                  .setEndpoint(endpoint.getUrl())
35                  .setClient(new ApacheClient(builder.build()))
36                  .setConverter(new IodConverter(new JacksonConverter()))
37                  .setErrorHandler(new IodErrorHandler());
38  
39  
40          if (withInterceptor) {
41              restAdapterBuilder.setRequestInterceptor(new ApiKeyRequestInterceptor(endpoint.getApiKey()));
42          }
43  
44          return restAdapterBuilder.build();
45      }
